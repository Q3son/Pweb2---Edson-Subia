Días para el Día de Arequipa

Calcula cuántos días faltan para el 15 de agosto

Código

function diasParaArequipa() {
    const hoy = new Date();
    const año = hoy.getFullYear();
    let diaArequipa = new Date(año, 7, 15); // 15 de agosto (mes 7)
    
    if (hoy > diaArequipa) {
        diaArequipa = new Date(año + 1, 7, 15);
    }
    
    const diff = diaArequipa - hoy;
    return Math.ceil(diff / (1000 * 60 * 60 * 24));
}
