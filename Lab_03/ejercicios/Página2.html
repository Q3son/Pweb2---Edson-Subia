Calculadora de Operaciones

Realiza operaciones aritméticas, lógicas y de bits

Código

function calcular(operador, a, b) {
    a = Number(a);
    b = Number(b);
    
    switch (operador) {
        case '+': return a + b;
        case '-': return a - b;
        case '*': return a * b;
        case '/': return a / b;
        case '%': return a % b;
        case '**': return a ** b;
        case '&': return a & b;
        case '|': return a | b;
        case '^': return a ^ b;
        case '<<': return a << b;
        case '>>': return a >> b;
        case '&&': return Boolean(a && b);
        case '||': return Boolean(a || b);
        default: return 'Operador no válido';
    }
}
